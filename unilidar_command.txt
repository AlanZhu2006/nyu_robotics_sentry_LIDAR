3.2 Activate navigation 2

连通 Nav2 和 LIO:
ros2 run tf2_ros static_transform_publisher 0 0 0 0 0 0 odom camera_init &
ros2 run tf2_ros static_transform_publisher 0 0 0 0 -0.873 0 aft_mapped base_link &
ros2 run tf2_ros static_transform_publisher 0 0 0 0 0 0 base_link unilidar_imu_initial &
ros2 run tf2_ros static_transform_publisher 0 0 0 0 0 0 base_link base_footprint &

运行转换节点 (注意 cloud_in 对应你的雷达话题):
ros2 run pointcloud_to_laserscan pointcloud_to_laserscan_node --ros-args \
-p target_frame:=base_link \
-p transform_tolerance:=0.01 \
-p min_height:=-0.4 \
-p max_height:=1.0 \
-p angle_min:=-3.1415 \
-p angle_max:=3.1415 \
-p range_min:=0.1 \
-p range_max:=20.0 \
-p use_inf:=true \
-r cloud_in:=/unilidar/cloud \
-r scan:=/scan

open nav2:
ros2 launch nav2_bringup bringup_launch.py \
    use_sim_time:=False \

open rviz:
ros2 run rviz2 rviz2 -d $(ros2 pkg prefix nav2_bringup)/share/nav2_bringup/rviz/nav2_default_view.rviz

3.2.1 autonomous navigation:(after nav2 goal)
订阅 /cmd_vel -> 提取数据 -> 通过串口发送给底盘
# 方式1: ROS2模式（推荐）- 自动订阅 /cmd_vel 话题并转发到底盘
sudo chmod 777 /dev/ttyACM0
cd ~/Desktop
source ~/nav_ws/install/setup.bash
python3 serial_sender.py --port /dev/ttyACM0 --baud 115200 --ros2

# 方式2: keyboard control（推荐）
sudo chmod 777 /dev/ttyACM0
cd ~/Desktop
source ~/nav_ws/install/setup.zsh
python3 serial_sender.py --port /dev/ttyACM0 --keyboard --speed 0.3

# 方式3: 直接发送速度命令（测试用）
sudo chmod 777 /dev/ttyACM0
cd ~/Desktop
source ~/nav_ws/install/setup.bash
python3 serial_sender.py --port /dev/ttyACM0 --vx -0.5 --vy 0.0 --wz 0.0 --duration 0.5
