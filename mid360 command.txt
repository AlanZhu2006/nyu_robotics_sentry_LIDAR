start with one command!!! (nav2)
----------------------------
~/nav_ws/start_robot.sh

1. 单雷达启动 (仅内置网口，雷达 192.168.1.182):
sudo ip addr add 192.168.1.2/24 dev enp114s0
cd ~/nav_ws
source install/setup.bash
ros2 launch livox_ros_driver2 msg_MID360_launch.py

1. 双雷达启动 (雷达1内置 182 + 雷达2 USB 114，双驱动+合并):
# 首次使用需先构建: cd ~/nav_ws && colcon build --packages-select livox_dual_merge && source install/setup.bash
# 步骤 1.1 网络与路由
sudo ip addr add 192.168.1.2/24 dev enp114s0
sudo ip addr add 192.168.1.3/24 dev enx00e04c2536b0
sudo ip route add 192.168.1.114/32 dev enx00e04c2536b0
sudo ip route add 192.168.1.182/32 dev enp114s0

# 步骤 1.2 启动双雷达（双驱动+合并，发布 /livox/lidar 和 /livox/imu）
cd ~/nav_ws
source install/setup.bash
ros2 launch livox_dual_merge dual_lidar_merge_launch.py

# 或一键运行: ./test_dual_lidar.sh

2. Release topic /Odometry:

export LD_PRELOAD=/lib/x86_64-linux-gnu/libusb-1.0.so.0
cd ~/nav_ws
source install/setup.bash
ros2 launch fast_lio mapping.launch.py config_file:=mid360.yaml


3.1 mapping and save (first map then nav)

see pointcloud:
export LD_PRELOAD=/lib/x86_64-linux-gnu/libusb-1.0.so.0
cd ~/nav_ws/src/FAST_LIO/PCD/
pcl_viewer scans.pcd

ROTATE pointcloud:(the address should be changed in pcd2pgm config depending on LIO)
cd ~/nav_ws/src/FAST_LIO/PCD
python3 rotate_pcd.py

3d PCD to 2d pgm:
export LD_PRELOAD=/lib/x86_64-linux-gnu/libusb-1.0.so.0
cd ~/nav_ws
source install/setup.bash
ros2 launch pcd2pgm pcd2pgm_launch.py

使用 nav2 工具保存:
cd ~/Desktop/map
ros2 run nav2_map_server map_saver_cli -f 11_map

3.2 Activate navigation 2

连通 Nav2 和 LIO:
ros2 run tf2_ros static_transform_publisher 0 0 0 0 0 0 odom camera_init &
ros2 run tf2_ros static_transform_publisher 0 0 0 0 -0.873 0 body base_link &
ros2 run tf2_ros static_transform_publisher 0 0 0 0 0 0 base_link base_footprint &

运行转换节点 (注意 cloud_in 对应你的雷达话题):
ros2 run pointcloud_to_laserscan pointcloud_to_laserscan_node --ros-args \
-p target_frame:=base_link \
-p transform_tolerance:=0.01 \
-p min_height:=-0.4 \
-p max_height:=1.0 \
-p angle_min:=-3.1415 \
-p angle_max:=3.1415 \
-p range_min:=0.1 \
-p range_max:=20.0 \
-p use_inf:=true \
-p qos_overrides./cloud_in.reliability:=best_effort \
-r cloud_in:=/cloud_registered \
-r scan:=/scan &

open nav2:
ros2 launch nav2_bringup bringup_launch.py \
    use_sim_time:=False \
    map:=/home/nyu/Desktop/map/11_map.yaml \
    params_file:=/home/nyu/nav_ws/my_nav2_params.yaml &
    
open rviz:
ros2 run rviz2 rviz2 -d $(ros2 pkg prefix nav2_bringup)/share/nav2_bringup/rviz/nav2_default_view.rviz

3.2.1 autonomous navigation:(after nav2 goal)
订阅 /cmd_vel -> 提取数据 -> 通过串口发送给底盘
# 方式1: ROS2模式（推荐）- 自动订阅 /cmd_vel 话题并转发到底盘
sudo chmod 777 /dev/ttyACM0
cd ~/Desktop
source ~/nav_ws/install/setup.bash
python3 serial_sender.py --port /dev/ttyACM0 --baud 115200 --ros2

# 方式2: keyboard control（推荐）
sudo chmod 777 /dev/ttyACM0
cd ~/Desktop
source ~/nav_ws/install/setup.zsh
python3 serial_sender.py --port /dev/ttyACM0 --keyboard --speed 1.0

# 方式3: 直接发送速度命令（测试用）
sudo chmod 777 /dev/ttyACM0
cd ~/Desktop
source ~/nav_ws/install/setup.bash
python3 serial_sender.py --port /dev/ttyACM0 --vx -0.5 --vy 0.0 --wz 0.0 --duration 0.1
